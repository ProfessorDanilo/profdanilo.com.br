#include <stdio.h>

int soma(int a, int b){
    return a+b;
}

int diferenca(int a, int b){
    return a - b;
}

int multiplicacao(int a, int b){
    return a*b;
}

int divisao(int a, int b){
    return a/b;
}

int resto(int a, int b){
    return a%b;
}

int quociente(int a, int b){
    return (a-resto(a, b))/b;
}

int main(){
    int valor1 = 7, valor2 = 3;
    printf("A soma é: %d\n", soma(valor1, valor2));
    printf("A diferênça é: %d\n", diferenca(valor1, valor2));
    printf("A multiplicação é: %d\n", multiplicacao(valor1, valor2));
    printf("A divisao é: %d\n", divisao(valor1, valor2));
    printf("O resto é: %d\n", resto(valor1, valor2));
    printf("O quociente é: %d", quociente(valor1, valor2));
    return 0;
}
